function renderTags(tags) { const tagCounts = {}; const recipes = getRecipes(); // Assuming this function retrieves the list of recipes // Count recipes for each tag recipes.forEach(recipe => { recipe.tags.forEach(tag => { if (!tagCounts[tag]) { tagCounts[tag] = 0; } tagCounts[tag]++; }); }); const tagContainer = document.getElementById('tag-container'); tagContainer.innerHTML = ''; // Clear existing tags // Render buttons with tag counts for (const tag in tagCounts) { const button = document.createElement('button'); button.innerText = `${tag} (${tagCounts[tag]})`; tagContainer.appendChild(button); } }
