ingredients:
  - "4 teaspoons olive oil"
  - "1 package chicken sausage"
  - "½ cup yellow onion"
  - "1½ cup orzo"
  - "6 teaspoons garlic"
  - "2½ cups chicken broth"
  - "¼ teaspoon salt"
  - "¼ teaspoon red pepper flakes"
  - "2 heads broccolini"
  - "¼ cup shredded parmesan cheese"
  - "2 teaspoons lemon zest"
ingredients_subhead: ""
ingredients_2:
  - ""
preparations:
  - "Chop the onion."
  - "Cut the chicken sausage into ¼-inch slices."
directions:
  - "Heat the oil in a large pot over medium-high heat."
  - "Add the sausage and onion; cook until the sausage is browned (3&nbsp;minutes)."
  - "Add the orzo and garlic; stir for 1 minute."
  - "Stir in the broth, salt, and red pepper."
  - "Bring to a boil."
  - "Add the broccolini."
  - "Return to a boil, then reduce heat and simmer covered until orzo is tender (about 8 minutes)."
  - "Continue cooking uncovered until the broth is absorbed."
  - "Stir in parmesan cheese and lemon zest."
